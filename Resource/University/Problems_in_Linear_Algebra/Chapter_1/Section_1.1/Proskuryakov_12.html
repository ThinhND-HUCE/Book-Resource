\(\begin{array}{ll} & \begin{vmatrix} \sin\alpha + \sin\beta & \cos\beta + \cos\alpha \\ \cos\beta - \cos\alpha & \sin\alpha - \sin\beta \end{vmatrix} \\ = & \left(\sin\alpha + \sin\beta\right)\left(\sin\alpha - \sin\beta\right) - \left( \cos\beta - \cos\alpha \right) \left( \cos\beta + \cos\alpha \right) \\ = & \left( \sin^2\alpha - \sin^2\beta \right) - \left( \cos^2\beta - \cos^2\alpha \right) \\ = & \left( \sin^2\alpha + \cos^2\alpha \right) - \left( \sin^2\beta + \cos^2\beta \right) = 1 - 1 = 0 \end{array}\)
