\(\begin{array}{ll} & \begin{vmatrix} 2\sin\varphi\cos\varphi & 2\sin^2\varphi - 1 \\ 2\cos^2\varphi - 1 & 2\sin\varphi\cos\varphi \end{vmatrix} = \begin{vmatrix} \sin 2\varphi & - \cos 2\varphi \\ \cos 2\varphi & \sin 2\varphi \end{vmatrix} \\ = & \sin 2\varphi\times \sin 2\varphi - \cos 2\varphi\times\left( -\cos 2\varphi \right) = \sin^2 2\varphi + \cos^2 2\varphi = 1 \end{array}\)
